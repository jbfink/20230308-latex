\begin{table}[h!] % h! eliminates float
\begin{center}
	\caption{Your first table.}
	\label{tab:table1}
	\begin{tabular}{l|c|r} % <-- Alignments: 1st column left, 2nd middle and 3rd right, with vertical lines in between
